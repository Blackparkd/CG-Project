158int main(int argc, char** argv) {
159 // Create a map to associate primitive names with handler functions
160 std::unordered_map<std::string, void (*)(int, char**)> shapeHandlers;
161 shapeHandlers["plane"] = handlePlane;
162 shapeHandlers["box"] = handleBox;
163 shapeHandlers["cone"] = handleCone;
164 shapeHandlers["sphere"] = handleSphere;
165
166 // Check if the user requested help
167 if (argc == 2 && (std::string(argv[1]) == "--help" || std::string(argv[1]) == "-h")) {
168 showUsage(argv[0]);
169 return 0;
170 }
171
172 // Check if the user provided at least the shape name
173 if (argc < 2) {
174 std::cerr << "Error: No shape specified.\n";
175 showUsage(argv[0]);
176 return 1;
177 }
178
179 // Get the shape name
180 std::string shape = argv[1];
181
182 // Check if the shape is valid
183 if (shapeHandlers.find(shape) == shapeHandlers.end()) {
184 std::cerr << "Error: Invalid shape '" << shape << "'.\n";
185 showUsage(argv[0]);
186 return 1;
187 }
188
189 // Call the appropriate handler function
190 try {
191 shapeHandlers[shape](argc, argv);
192 } catch (const std::exception& e) {
193 std::cerr << e.what() << "\n";
194 showUsage(argv[0]);
195 return 1;
196 }
197
198 return 0;
199}
