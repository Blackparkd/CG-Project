64void renderScene() {
65 // Clear the color and depth buffers
66 glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
67
68 // Reset the modelview matrix
69 glLoadIdentity();
70
71 // Set up camera position, look-at point, and up vector
72 gluLookAt(world.camera.position.x, world.camera.position.y, world.camera.position.z,
73 world.camera.lookAt.x, world.camera.lookAt.y, world.camera.lookAt.z,
74 world.camera.up.x, world.camera.up.y, world.camera.up.z);
75
76 // Draw each model listed in the XML
77 for (const Model& model : world.models) {
78 glBegin(GL_TRIANGLES); // Drawing triangles for 3D models
79 const std::vector<Point>& vertices = model.vertices;
80
81 // Calculate slices from vertex count (for debugging purposes)
82 int slices = sqrt(vertices.size()) - 1;
83
84 // Process vertices in groups of 3 to form triangles
85 for (size_t i = 0; i < vertices.size(); i += 3) {
86 glVertex3f(vertices[i].x, vertices[i].y, vertices[i].z);
87 glVertex3f(vertices[i + 1].x, vertices[i + 1].y, vertices[i + 1].z);
88 glVertex3f(vertices[i + 2].x, vertices[i + 2].y, vertices[i + 2].z);
89 }
90 glEnd();
91 }
92
93 // Swap front and back buffers (double buffering)
94 glutSwapBuffers();
95}

106void changeSize(int w, int h) {
